\newcommand{\tra}[1]{%
  \ifcase#1\relax%
  \or 壹% 1
  \or 贰% 2
  \or 叁% 3
  \or 肆% 4
  \or 伍% 5
  \or 陆% 6

  \or 捌% 8

  \or 拾% 10

}
